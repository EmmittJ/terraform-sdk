check "multi_source_check" {
  data "http" "endpoint_a" {
    url = "https://a.example.com/health"
  }

  data "http" "endpoint_b" {
    url = "https://b.example.com/health"
  }

  assert {
    condition = data.http.endpoint_a.status_code == 200
    error_message = "Endpoint A health check failed"
  }

  assert {
    condition = data.http.endpoint_b.status_code == 200
    error_message = "Endpoint B health check failed"
  }

}
